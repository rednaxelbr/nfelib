bool chaveNfeValida(String chnfe) {
  if (chnfe.length != 44) {
    return false;
  }

  final _dv = int.tryParse(chnfe[43]);
  if (_dv == null) {
    return false;
  }

  return _dv == dvChaveNFe(chnfe);
}
